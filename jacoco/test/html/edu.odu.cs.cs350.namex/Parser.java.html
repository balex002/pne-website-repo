package edu.odu.cs.cs350.namex;

import java.util.Arrays;
import java.util.List;
import java.util.ArrayList;

public class Parser {
    public static List<String> tokenizeInput(String toTokenize){
        /* Break down the input so it can be analyzed by Machines.
        @return: A list of the tokens delimited by a space (strings)*/

        /*replace1() replaces the <NER> with an empty character so the tokenizer can parse the actual string.
          replace2() the </NER> with an empty string so the tokenizer does not token the NER tag.*/
        String replace1 = toTokenize.replaceAll("<NER>", "");
        String replace2 = replace1.replaceAll("</NER>", "");
        
        ///Separates the words in the string by a space and adds the word to a list.
        List<String> tokens = Arrays.asList(replace2.split(" "));
        
        return tokens;
    }
    
    public static List<String> shingleTokens(List<String> tokens){
        /* Merges tokens together with their neighbors. 
        Visit Trello user story or design notes for details.
        @return: A list of shingled tokens (strings )*/

        /*Create an empty list for the shingled tokens*/
        List<String> shingledTokens = new ArrayList<String>();

        /*Concatenate the strings from the list of  tokens with a space and then add the new shingles into shingledTokens*/
        for (int i = 0; i < tokens.size() - 1; i++){
            shingledTokens.add(i, (tokens.get(i) + " " + tokens.get(i + 1)));
        }

        return shingledTokens;
    }
}
